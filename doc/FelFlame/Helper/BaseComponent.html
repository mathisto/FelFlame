def new(**opts)
  new_component = super

  # Generate ID
  new_id = self.data.find_index { |i| i.nil? }
  new_id = self.data.size if new_id.nil?
  new_component.id = new_id

  # Fill params
  opts.each do |key, value|
    new_component.send "#{key}=", value
  end

  # Save Component
  data[new_id] = new_component
end
